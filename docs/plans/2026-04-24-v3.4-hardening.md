```swift
/// Called when an initial message arrives from a peer not in TrustedContactStore.
/// Receiver must show a confirmation UI and call `respondToSession` only on approval.
var onFirstContactConsentRequired: ((_ peerId: String, _ fingerprint: String) -> Void)?
private var pendingFirstContact: [String: InitialEnvelope] = [:]

func handleIncoming(envelope: RemoteMessageEnvelope) async throws {
    // Existing decrypt path
    if session(for: envelope.senderId) != nil {
        // Active session — proceed normally
        return try await decryptInSession(envelope)
    }
    // First contact: check trust store
    if let contact = trustedContacts.contact(for: envelope.senderId),
       contact.trustLevel != .unknown {
        // Known peer — proceed directly
        return try await respondToSession(envelope)
    }
    // Unknown peer: defer and surface consent
    pendingFirstContact[envelope.senderId] = envelope
    let fp = fingerprintString(of: envelope.theirIdentityKey)
    onFirstContactConsentRequired?(envelope.senderId, fp)
}

func approveFirstContact(_ peerId: String) async throws {
    guard let envelope = pendingFirstContact.removeValue(forKey: peerId) else { return }
    trustedContacts.add(TrustedContact(id: peerId, publicKey: envelope.theirIdentityKey, trustLevel: .user_approved))
    try await respondToSession(envelope)
}

func rejectFirstContact(_ peerId: String) {
    pendingFirstContact.removeValue(forKey: peerId)
}
```

**Step 3: Implement**

Replace `consumeOneTimePreKey` with transactional version:

```swift
/// Persisted set of OTP IDs that have already been consumed. Checked on load
/// so that a crash-after-remove-before-save cannot re-expose a consumed key.
private var consumedOTPIds: Set<UUID> = []

func consumeOneTimePreKey(id: UUID, deferSave: Bool = false) -> OneTimePreKey? {
    // Reject already-consumed IDs, even if still in the map (never happens
    // under normal flow but guards against concurrent double-consume races).
    guard !consumedOTPIds.contains(id) else { return nil }
    guard let otp = oneTimePreKeys.first(where: { $0.id == id }) else { return nil }

    // Atomic: record consumption FIRST, then persist, then remove.
    consumedOTPIds.insert(id)
    if !deferSave { saveConsumedSet() }
    oneTimePreKeys.removeAll { $0.id == id }
    if !deferSave { save() }
    return otp
}

private func saveConsumedSet() {
    let encoded = try? JSONEncoder().encode(consumedOTPIds)
    if let encoded {
        keychainStore.save(key: "peerDropConsumedOTPs", data: encoded)
    }
}

private func loadConsumedSet() {
    guard let data = keychainStore.load(key: "peerDropConsumedOTPs"),
          let decoded = try? JSONDecoder().decode(Set<UUID>.self, from: data) else {
        consumedOTPIds = []
        return
    }
    consumedOTPIds = decoded
    // Evict any OTPs that are also in the consumed set (crash recovery path).
    oneTimePreKeys.removeAll { consumedOTPIds.contains($0.id) }
}
```

Call `loadConsumedSet()` from init.

**Step 1: Design**

`RelaySession` is an actor that owns the full lifecycle of a single relay attempt:
- ICE config acquisition
- DataChannelClient creation
- WorkerSignaling lifecycle (open, close)
- SDP offer/answer exchange
- ICE candidate relay
- DataChannelTransport wiring
- Phase timers (1→2 at 8s, 3 at 30s)
- RelayHintsStore update on completion
- ConnectionMetrics token lifecycle

**Critical invariant enforced by `deinit`**: if the session ends in any state other than `.connected`, its `signaling.disconnect()` is called. This is the structural fix that makes the v3.3.0 bug **impossible**, not just unlikely.

```swift
actor RelaySession {
    enum Role { case creator, joiner(roomToken: String) }
    enum Outcome { case connected(DataChannelTransport), failed(Error), abandoned }

    private let roomCode: String
    private let role: Role
    private let signaling: WorkerSignaling
    private var client: DataChannelClient?
    private var metricsToken: ConnectionMetrics.Token?
    private var outcome: Outcome = .abandoned
    private var state: State = .pending

    // Observers (MainActor-dispatched)
    @MainActor var onConnected: ((DataChannelTransport) -> Void)?
    @MainActor var onFailed: ((Error) -> Void)?
    @MainActor var onFirstContactRequired: ((PeerIdentity) -> Void)?

    init(roomCode: String, role: Role, signaling: WorkerSignaling) {
        self.roomCode = roomCode
        self.role = role
        self.signaling = signaling
    }

    deinit {
        // Structural guarantee: signaling is closed if session ends unsucessfully.
        if case .connected = outcome { return }
        Task { await signaling.disconnect() }
    }

    func start() async {
        // ... ICE, SDP, wiring, phase timers, all in one place
    }

    func cancel() async {
        outcome = .abandoned
        await signaling.disconnect()
        client?.close()
    }
}
```
